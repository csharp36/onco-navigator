class EncryptionConverterTest {
    
    private EncryptionConverter converter;
    
    @BeforeEach
    void setUp() {
        // Set up with a known test key (256-bit AES key)
        byte[] keyBytes = new byte[32]; // 256 bits
        new java.security.SecureRandom().nextBytes(keyBytes);
        SecretKey testKey = new SecretKeySpec(keyBytes, "AES");
        
        // Use reflection or test-specific constructor to inject key
        // The converter needs the key from Spring context - for unit testing,
        // create a subclass or use a test factory
        converter = new EncryptionConverter(testKey); // Add constructor that accepts key
    }
    
    @Test
    void encryptDecrypt_roundTrip_returnsOriginal() {
        String original = "John Smith";
        byte[] encrypted = converter.convertToDatabaseColumn(original);
        String decrypted = converter.convertToEntityAttribute(encrypted);
        assertEquals(original, decrypted);
    }
    
    @Test
    void encryptDecrypt_emptyString_returnsEmpty() {
        String original = "";
        byte[] encrypted = converter.convertToDatabaseColumn(original);
        String decrypted = converter.convertToEntityAttribute(encrypted);
        assertEquals(original, decrypted);
    }
    
    @Test
    void encrypt_nullValue_returnsNull() {
        assertNull(converter.convertToDatabaseColumn(null));
    }
    
    @Test
    void decrypt_nullValue_returnsNull() {
        assertNull(converter.convertToEntityAttribute(null));
    }
    
    @Test
    void encrypt_sameInput_producesDifferentCiphertext() {
        // AES-GCM with random IV should produce different ciphertext each time
        String input = "Patient Name";
        byte[] encrypted1 = converter.convertToDatabaseColumn(input);
        byte[] encrypted2 = converter.convertToDatabaseColumn(input);
        assertFalse(java.util.Arrays.equals(encrypted1, encrypted2),
            "Same plaintext must produce different ciphertext (random IV)");
    }
    
    @Test
    void encrypt_outputContainsIvPlusCiphertext() {
        String input = "test";
        byte[] encrypted = converter.convertToDatabaseColumn(input);
        // 12 bytes IV + at least some ciphertext + 16 bytes GCM tag
        assertTrue(encrypted.length >= 12 + 1 + 16,
            "Encrypted output must contain IV (12) + ciphertext + GCM tag (16)");
    }
}

        AuditLogEntry saved = auditLogRepository.save(entry);
        assertNotNull(saved.getId());
    }
    
    @Test
    void update_auditEntry_throwsException() {
        // Verify that UPDATE is rejected at database level
        AuditLogEntry entry = createTestEntry();
        AuditLogEntry saved = auditLogRepository.save(entry);
        
        // Direct JDBC UPDATE should fail due to permission restrictions
        // Note: This test verifies the V3 migration permission setup

            );
        });
    }
    
    @Test
    void delete_auditEntry_throwsException() {
        AuditLogEntry entry = createTestEntry();
        AuditLogEntry saved = auditLogRepository.save(entry);
        
        assertThrows(DataAccessException.class, () -> {
            jdbcTemplate.update("DELETE FROM audit_log WHERE id = ?", saved.getId());
        });
    }
    
    @Test
    void findByActorIdAndTimestampBetween_returnsEntries() {
        UUID actorId = UUID.randomUUID();
        AuditLogEntry entry = createTestEntry();

        auditLogRepository.save(entry);

            actorId,
            OffsetDateTime.now().minusHours(1),
            OffsetDateTime.now().plusHours(1)

WORKDIR /app

# Health check
HEALTHCHECK --interval=30s --timeout=10s --start-period=60s --retries=3 \
