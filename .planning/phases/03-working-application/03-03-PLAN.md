```java
    private final PatientRepository patientRepository;

    //   1. Load patient or 404
    //   2. Load template: templateRepository.findByCancerType(patient.getCancerType()).orElseThrow(NOT_FOUND)
    //   3. Parse JSONB: List<PathwayStep> steps = objectMapper.readValue(template.getTemplateData(), new TypeReference<>() {})
    //   4. Load care events: careEventRepository.findByPatient_IdOrderByEventDateDesc(patientId)
    //   5. Load open alerts: alertRepository.findByPatientIdAndStatus(patientId, AlertStatus.OPEN)
    //   6. For each PathwayStep: call deriveStepStatus(step, careEvents, openAlerts, patient.getDiagnosisDate())
    //   7. Return new PathwayStatusResponse(patientId, stepStatuses)

    // deriveStepStatus(PathwayStep step, List<CareEvent> events, List<Alert> openAlerts, LocalDate diagnosisDate) -> PathwayStepStatus

    //   Check hasActiveAlert: openAlerts.stream().anyMatch(a -> a.getPathwayStepName().equals(step.name()))
    //   Build timingInfo string: "Completed Day X of Y-day window" or "Z days overdue" or "Waiting for Step N"
```

```java
@RestController
@RequestMapping("/api/patients")
public class PatientController {

    private final PatientService patientService;
    private final PathwayStatusService pathwayStatusService;

    // Constructor injection

    @PostMapping
    @PreAuthorize("hasRole('CARE_COORDINATOR') or hasRole('ADMIN')")
    @ResponseStatus(HttpStatus.CREATED)
    public PatientResponse createPatient(
            @Valid @RequestBody CreatePatientRequest request,
            @AuthenticationPrincipal Jwt jwt) {
        UUID actorId = UUID.fromString(jwt.getSubject());
        return patientService.createPatient(request, actorId);
    }

    @GetMapping
    @PreAuthorize("hasRole('CARE_COORDINATOR') or hasRole('NURSE_NAVIGATOR') or hasRole('ADMIN')")
    public List<PatientResponse> listPatients(
            @RequestParam(required = false) String mrn) {
        if (mrn != null && !mrn.isBlank()) {
            return patientService.findByMrn(mrn);
        }
        return patientService.findAll();
    }

    @GetMapping("/{id}")
    @PreAuthorize("hasRole('CARE_COORDINATOR') or hasRole('NURSE_NAVIGATOR') or hasRole('ADMIN')")
    public PatientResponse getPatient(@PathVariable UUID id) {
        return patientService.findById(id);
    }

    @PatchMapping("/{id}/deactivate")
    @PreAuthorize("hasRole('CARE_COORDINATOR') or hasRole('ADMIN')")
    @ResponseStatus(HttpStatus.NO_CONTENT)
    public void deactivatePatient(@PathVariable UUID id,
                                  @Valid @RequestBody DeactivatePatientRequest request,
                                  @AuthenticationPrincipal Jwt jwt) {
        UUID actorId = UUID.fromString(jwt.getSubject());
        patientService.deactivatePatient(id, request, actorId);
    }

    @GetMapping("/{id}/pathway-status")
    @PreAuthorize("hasRole('CARE_COORDINATOR') or hasRole('NURSE_NAVIGATOR') or hasRole('ADMIN')")
    public PathwayStatusResponse getPathwayStatus(@PathVariable UUID id) {
        return pathwayStatusService.getPathwayStatus(id);
    }
}
```

**DashboardController.java** — routes through PatientService (not PatientRepository directly):
```java
@RestController
@RequestMapping("/api/dashboard")
public class DashboardController {

    private final AlertService alertService;
    private final PatientService patientService;

    // Constructor injection — PatientService (not PatientRepository)

    @GetMapping("/stats")
    @PreAuthorize("isAuthenticated()")
    public DashboardStatsResponse getStats() {
        long openAlertCount = alertService.countOpenAlerts();
        long activePatients = patientService.countActivePatients();

        List<AlertResponse> allAlerts = alertService.getOpenAlerts();
        long patientsWithAlerts = allAlerts.stream()
            .map(AlertResponse::patientId)
            .distinct()
            .count();
        long onTrackPatients = activePatients - patientsWithAlerts;

        List<AlertResponse> topUrgent = allAlerts.stream().limit(5).toList();

        return new DashboardStatsResponse(openAlertCount, activePatients,
            Math.max(0, onTrackPatients), topUrgent);
    }
}
```

**RBAC rules (CRITICAL — from RESEARCH.md Pitfall 7):**
- Use `hasRole('CARE_COORDINATOR')` — Spring prepends ROLE_ automatically
- NEVER use `hasRole('ROLE_CARE_COORDINATOR')` — would check for ROLE_ROLE_CARE_COORDINATOR which never matches
- Write endpoints (POST, PATCH): `hasRole('CARE_COORDINATOR') or hasRole('ADMIN')`
- Alert management (list, resolve): `hasRole('NURSE_NAVIGATOR') or hasRole('ADMIN')`
- Read-only endpoints (GET patients, GET patient/{id}, pathway-status): `hasRole('CARE_COORDINATOR') or hasRole('NURSE_NAVIGATOR') or hasRole('ADMIN')` — ensures only clinical staff roles can access patient data (mitigates BOLA per T-03-08)
- Alert count (lightweight polling): `isAuthenticated()` (all roles need badge)

All controllers use `@AuthenticationPrincipal Jwt jwt` and `UUID.fromString(jwt.getSubject())` to extract the actor identity for audit purposes.
